152 /**
153  * @brief This function handles TIM4 global interrupt request.
154  * @param None
155  * @retval None
156  */
157 void TIM4_IRQHandler(void)
158 {
159  RCC_ClocksTypeDef RCC_Clocks;
160  RCC_GetClocksFreq(&RCC_Clocks);
161 
162  /* Clear TIM4 Capture compare interrupt pending bit */
163  TIM_ClearITPendingBit(TIM4, TIM_IT_CC2);
164 
165  /* Get the Input Capture value */
166  IC2Value = TIM_GetCapture2(TIM4);
167 
168  if (IC2Value != 0)
169  {
170  /* Duty cycle computation */
171  DutyCycle = (TIM_GetCapture1(TIM4) * 100) / IC2Value;
172 
173  /* Frequency computation
174  TIM4 counter clock = (RCC_Clocks.HCLK_Frequency)/2 */
175 
176  Frequency = (RCC_Clocks.HCLK_Frequency)/2 / IC2Value;
177  }
178  else
179  {
180  DutyCycle = 0;
181  Frequency = 0;
182  }
183 }
