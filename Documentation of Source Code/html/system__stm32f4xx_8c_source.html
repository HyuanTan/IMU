202 /**
203  * @brief Setup the microcontroller system
204  * Initialize the Embedded Flash Interface, the PLL and update the
205  * SystemFrequency variable.
206  * @param None
207  * @retval None
208  */
209 void SystemInit(void)
210 {
211  /* FPU settings ------------------------------------------------------------*/
212  #if (__FPU_PRESENT == 1) && (__FPU_USED == 1)
213  SCB->CPACR |= ((3UL << 10*2)|(3UL << 11*2)); /* set CP10 and CP11 Full Access */
214  #endif
215 
216  /* Reset the RCC clock configuration to the default reset state ------------*/
217  /* Set HSION bit */
218  RCC->CR |= (uint32_t)0x00000001;
219 
220  /* Reset CFGR register */
221  RCC->CFGR = 0x00000000;
222 
223  /* Reset HSEON, CSSON and PLLON bits */
224  RCC->CR &= (uint32_t)0xFEF6FFFF;
225 
226  /* Reset PLLCFGR register */
227  RCC->PLLCFGR = 0x24003010;
228 
229  /* Reset HSEBYP bit */
230  RCC->CR &= (uint32_t)0xFFFBFFFF;
231 
232  /* Disable all interrupts */
233  RCC->CIR = 0x00000000;
234 
235 #ifdef DATA_IN_ExtSRAM
236  SystemInit_ExtMemCtl();
237 #endif /* DATA_IN_ExtSRAM */
238 
239  /* Configure the System clock source, PLL Multiplier and Divider factors,
240  AHB/APBx prescalers and Flash settings ----------------------------------*/
241  SetSysClock();

251 /**
252  * @brief Update SystemCoreClock variable according to Clock Register Values.
253  * The SystemCoreClock variable contains the core clock (HCLK), it can
254  * be used by the user application to setup the SysTick timer or configure
255  * other parameters.
256  *
257  * @note Each time the core clock (HCLK) changes, this function must be called
258  * to update SystemCoreClock variable value. Otherwise, any configuration
259  * based on this variable will be incorrect.
260  *
261  * @note - The system frequency computed by this function is not the real
262  * frequency in the chip. It is calculated based on the predefined
263  * constant and the selected clock source:
264  *
265  * - If SYSCLK source is HSI, SystemCoreClock will contain the HSI_VALUE(*)
266  *
267  * - If SYSCLK source is HSE, SystemCoreClock will contain the HSE_VALUE(**)
268  *
269  * - If SYSCLK source is PLL, SystemCoreClock will contain the HSE_VALUE(**)
270  * or HSI_VALUE(*) multiplied/divided by the PLL factors.
271  *
272  * (*) HSI_VALUE is a constant defined in stm32f4xx.h file (default value
273  * 16 MHz) but the real value may vary depending on the variations
274  * in voltage and temperature.
275  *
276  * (**) HSE_VALUE is a constant defined in stm32f4xx.h file (default value
277  * 25 MHz), user has to ensure that HSE_VALUE is same as the real
278  * frequency of the crystal used. Otherwise, this function may
279  * have wrong result.
280  *
281  * - The result of this function could be not correct when using fractional
282  * value for HSE crystal.
283  *
284  * @param None
285  * @retval None
286  */
287 void SystemCoreClockUpdate(void)
288 {
289  uint32_t tmp = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;
290 
291  /* Get SYSCLK source -------------------------------------------------------*/
292  tmp = RCC->CFGR & RCC_CFGR_SWS;
293 
294  switch (tmp)
295  {
296  case 0x00: /* HSI used as system clock source */
297  SystemCoreClock = HSI_VALUE;
298  break;
299  case 0x04: /* HSE used as system clock source */
300  SystemCoreClock = HSE_VALUE;
301  break;
302  case 0x08: /* PLL used as system clock source */
303 
304  /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
305  SYSCLK = PLL_VCO / PLL_P
306  */
307  pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
308  pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
309 
310  if (pllsource != 0)
311  {
312  /* HSE used as PLL clock source */
313  pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
314  }
315  else
316  {
317  /* HSI used as PLL clock source */
318  pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
319  }
320 
321  pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >>16) + 1 ) *2;
322  SystemCoreClock = pllvco/pllp;
323  break;
324  default:
325  SystemCoreClock = HSI_VALUE;
326  break;
327  }
328  /* Compute HCLK frequency --------------------------------------------------*/
329  /* Get HCLK prescaler */
330  tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
331  /* HCLK frequency */
332  SystemCoreClock >>= tmp;
333 }

335 /**
336  * @brief Configures the System clock source, PLL Multiplier and Divider factors,
337  * AHB/APBx prescalers and Flash settings
338  * @Note This function should be called only once the RCC clock configuration
339  * is reset to the default reset state (done in SystemInit() function).
340  * @param None
341  * @retval None
342  */
343 static void SetSysClock(void)
344 {
345 /******************************************************************************/
346 /* PLL (clocked by HSE) used as System clock source */
347 /******************************************************************************/
348  __IO uint32_t StartUpCounter = 0, HSEStatus = 0;
349 
350  /* Enable HSE */
351  RCC->CR |= ((uint32_t)RCC_CR_HSEON);
352 
353  /* Wait till HSE is ready and if Time out is reached exit */
354  do
355  {
356  HSEStatus = RCC->CR & RCC_CR_HSERDY;
357  StartUpCounter++;
358  } while((HSEStatus == 0) && (StartUpCounter != HSE_STARTUP_TIMEOUT));
359 
360  if ((RCC->CR & RCC_CR_HSERDY) != RESET)
361  {
362  HSEStatus = (uint32_t)0x01;
363  }
364  else
365  {
366  HSEStatus = (uint32_t)0x00;
367  }
368 
369  if (HSEStatus == (uint32_t)0x01)
370  {
371  /* Select regulator voltage output Scale 1 mode, System frequency up to 168 MHz */
372  RCC->APB1ENR |= RCC_APB1ENR_PWREN;
373  PWR->CR |= PWR_CR_VOS;
374 
375  /* HCLK = SYSCLK / 1*/
376  RCC->CFGR |= RCC_CFGR_HPRE_DIV1;
377 
378  /* PCLK2 = HCLK / 2*/
379  RCC->CFGR |= RCC_CFGR_PPRE2_DIV2;
380 
381  /* PCLK1 = HCLK / 4*/
382  RCC->CFGR |= RCC_CFGR_PPRE1_DIV4;
383 
384  /* Configure the main PLL */
385  RCC->PLLCFGR = PLL_M | (PLL_N << 6) | (((PLL_P >> 1) -1) << 16) |
386  (RCC_PLLCFGR_PLLSRC_HSE) | (PLL_Q << 24);
387 
388  /* Enable the main PLL */
389  RCC->CR |= RCC_CR_PLLON;
390 
391  /* Wait till the main PLL is ready */
392  while((RCC->CR & RCC_CR_PLLRDY) == 0)
393  {
394  }
395 
396  /* Configure Flash prefetch, Instruction cache, Data cache and wait state */
397  FLASH->ACR = FLASH_ACR_ICEN |FLASH_ACR_DCEN |FLASH_ACR_LATENCY_5WS;
398 
399  /* Select the main PLL as system clock source */
400  RCC->CFGR &= (uint32_t)((uint32_t)~(RCC_CFGR_SW));
401  RCC->CFGR |= RCC_CFGR_SW_PLL;
402 
403  /* Wait till the main PLL is used as system clock source */
404  while ((RCC->CFGR & (uint32_t)RCC_CFGR_SWS ) != RCC_CFGR_SWS_PLL);
405  {
406  }
407  }
408  else
409  { /* If HSE fails to start-up, the application will have wrong clock
410  configuration. User can add here some code to deal with this error */
411  }
412 
413 }
